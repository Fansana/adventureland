function clone(d, b) {

        e.setTime(d.getTime());
        return e
    }
    if (d instanceof Array) {
        b.seen.push(d);
        var e = [];
        for (var c = 0; c < d.length; c++) {
            e[c] = clone(d[c], b)
        }
        return e
    }
    if (d instanceof Object) {
        b.seen.push(d);
        var e = {};
        for (var a in d) {
            if (d.hasOwnProperty(a)) {
                if (b.seen.indexOf(d[a]) !== -1) {
                    e[a] = 'circular_attribute[clone]';
